\subsection{CSS Grid Implementation}

\begin{tcolorbox}[colback=secondaryOrange!5, colframe=secondaryOrange, boxsep=5pt, title=\textbf{Grid Layout Code}]
Course Cards Grid:
\begin{lstlisting}[language=CSS, basicstyle=\ttfamily\small]
.course-grid {
  display: grid;
  grid-template-columns: repeat(auto-fit, minmax(300px, 1fr));
  gap: 2rem;
  padding: 2rem;
}
\end{lstlisting}
\end{tcolorbox}

\subsection{CSS Flexbox Implementation}

\begin{tcolorbox}[colback=accentGreen!5, colframe=accentGreen, boxsep=5pt, title=\textbf{Flexbox Layout Code}]
Navigation & Hero:
\begin{lstlisting}[language=CSS, basicstyle=\ttfamily\small]
nav { display: flex; justify-content: space-between; }
.hero { display: flex; flex-direction: column; 
        align-items: center; }
\end{lstlisting}
\end{tcolorbox}

\subsection{Responsive Media Queries}

\begin{tcolorbox}[colback=lightBg, colframe=darkGray, boxsep=5pt]

/* Tablet: 768px and above */
@media (max-width: 768px) { /* Adjust for tablet */ }

/* Mobile: 600px and above */
@media (max-width: 600px) { /* Adjust for mobile */ }
\end{lstlisting}
\end{tcolorbox}

\section{\textcolor{primaryBlue}{üí° Animation Examples}}

\begin{tcolorbox}[colback=secondaryOrange!5, colframe=secondaryOrange, boxsep=5pt]
\textbf{Hover Card Effect:}
\begin{lstlisting}[language=CSS, basicstyle=\ttfamily\small]
.card {
  transition: transform 0.3s ease, 
              box-shadow 0.3s ease;
}
.card:hover {
  transform: translateY(-10px);
  box-shadow: 0 10px 25px rgba(0,0,0,0.1);
}
\end{lstlisting}

\textbf{Keyframe Animation:}
\begin{lstlisting}[language=CSS, basicstyle=\ttfamily\small]
@keyframes slideIn {
  from { opacity: 0; transform: translateX(-20px); }
  to { opacity: 1; transform: translateX(0); }
}
.hero { animation: slideIn 0.8s ease-out; }
\end{lstlisting}
\end{tcolorbox}

\section{\textcolor{primaryBlue}{ü§ù Troubleshooting Guide}}

\begin{table}[h]
\centering
\begin{tabular}{|>{\columncolor{primaryBlue!10}}m{3.5cm}|m{4cm}|}
\hline
\textbf{Common Issues} & \textbf{Solution} \\
\hline
Grid items not aligning & Check \texttt{grid-template-columns} \& use \texttt{minmax()} \\
\hline
Mobile view breaking & Verify media queries in correct order (mobile-first) \\
\hline
Fonts not loading & Ensure Google Fonts link in \texttt{<head>} before CSS \\
\hline
Animations stuttering & Use \texttt{will-change} property, check performance \\
\hline
Images too large & Optimize images, use \texttt{max-width: 100\%} \\
\hline
Form validation failing & Add required attributes \& proper input types \\
\hline
\end{tabular}
